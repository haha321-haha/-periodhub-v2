// 类型定义（用于TypeScript类型检查）
export type AnchorTextSection = keyof typeof ANCHOR_TEXTS;
export type AnchorTextKey<T extends AnchorTextSection> = keyof typeof ANCHOR_TEXTS[T];

/**
 * 获取锚文本的辅助函数
 * @param section - 页面区域
 * @param key - 具体的锚文本键
 * @returns 优化后的锚文本
 */
export function getAnchorText<T extends AnchorTextSection>(
  section: T,
  key: AnchorTextKey<T>
): string {
  return ANCHOR_TEXTS[section][key] as string;
}

  variants: {
    homepage: {

      variant: '痛经管理指南'         // 优化版（实验组）

      variant: '经期健康指南'
    },
    scenarios: {
      office: {
        control: '了解更多',
        variant: '职场痛经应对'
      }
    }
  },
  
  /**
   * 获取A/B测试锚文本
   * @param section - 页面区域
   * @param userGroup - 用户分组（'control' 或 'variant'）
   */
  getTestAnchorText(section: string, userGroup: 'control' | 'variant'): string {
    if (!this.enabled) {
      return getAnchorText(section as AnchorTextSection, 'main' as any);
    }
    
    const variant = this.variants[section as keyof typeof this.variants];
    if (!variant) {
      return getAnchorText(section as AnchorTextSection, 'main' as any);
    }
    
    return variant[userGroup];
  }
};

// lib/linkGenerator.ts - 链接生成工具
import type { Locale } from '@/types/i18n';

interface LinkConfig {
  href: string;
  anchorTextKey: string;
  section: string;
}

export function generateOptimizedLink(
  config: LinkConfig,
  locale: Locale,
  anchorT: (key: string) => string
) {
  return {
    href: `/${locale}${config.href}`,
    text: anchorT(`${config.section}.${config.anchorTextKey}`)
  };
}

links.map(link => {

// components/TrackedLink.tsx - 带分析追踪的链接组件
'use client';

import Link from 'next/link';
import { useTranslations } from 'next-intl';

interface TrackedLinkProps {
  href: string;
  anchorTextKey: string;
  section: string;
  className?: string;
}

export function TrackedLink({ 
  href, 
  anchorTextKey, 
  section,
  className 
}: TrackedLinkProps) {
  const anchorT = useTranslations('anchorTexts');
  const anchorText = anchorT(`${section}.${anchorTextKey}`);
  
  const handleClick = () => {
    // 追踪链接点击
    if (typeof window !== 'undefined' && window.gtag) {
      window.gtag('event', 'anchor_click', {
        event_category: 'Internal Link',
        event_label: anchorText,
        page_path: window.location.pathname,
        destination: href
      });
    }
  };
  
  return (
    <Link 
      href={href} 
      className={className}
      onClick={handleClick}
    >
      {anchorText}
    </Link>
  );
}

## 常见问题

### Q1: 修改后出现TypeScript类型错误

**问题**: `Property 'anchorTexts' does not exist`

**解决方案**:
```tsx
// 确保正确导入
import { useTranslations } from 'next-intl';

// 客户端组件
const anchorT = useTranslations('anchorTexts');

// 服务端组件
const anchorT = await getTranslations('anchorTexts');
```

### Q2: 某些页面找不到通用锚文本

**原因**: 该页面可能已经使用了不同的文案

**解决方案**: 
1. 手动检查页面内容

3. 确认是否有其他需要优化的锚文本

### Q3: 中英文切换后锚文本没变化

**原因**: `messages/en.json` 未正确配置

**解决方案**:
```bash
# 检查英文配置文件
cat messages/en.json | grep anchorTexts

# 确保包含完整的anchorTexts配置
```

---

### Q4: 构建失败显示"Module not found"

**原因**: 导入路径错误

**解决方案**:
```tsx
// 正确的导入方式
import { useTranslations } from 'next-intl';           // 客户端
import { getTranslations } from 'next-intl/server';    // 服务端

// 错误的导入方式
import { useTranslations } from 'next-intl/server';    // ❌
```

---

### Q5: 服务端组件和客户端组件混淆

**区分方法**:
```tsx
// 客户端组件（有 'use client' 指令）
'use client';
import { useTranslations } from 'next-intl';

export default function ClientComponent() {
  const anchorT = useTranslations('anchorTexts');
  return <Link>{anchorT('homepage.main')}</Link>;
}

// 服务端组件（默认，无 'use client'）
import { getTranslations } from 'next-intl/server';

export default async function ServerComponent() {
  const anchorT = await getTranslations('anchorTexts');
  return <Link>{anchorT('homepage.main')}</Link>;
}
```
