// utils/metaGenerator.js
export async function generateMetaDesc(pageType, content) {
  const templates = {
    article: `${content.title}｜${content.excerpt} 了解更多>>`,
    tool: `【实用工具】${content.name}：${content.features.join('｜')}，立即体验`,
    guide: `${content.title}：${content.solutions.join('+')}方案｜${content.author}审核`
  };
  
  return templates[pageType] || content.description;
}

// 页面使用示例
export async function generateMetadata({ params }) {
  const guide = await getGuide(params.slug);
  return {
    description: generateMetaDesc('guide', {
      title: guide.title,
      solutions: guide.tags,
      author: '三甲医院妇科主任'
    })
  }
}
